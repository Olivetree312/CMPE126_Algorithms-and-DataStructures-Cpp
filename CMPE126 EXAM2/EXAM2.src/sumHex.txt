//HELPER FUNCTION TO CONVERT HEX CHAR TO INT
//works because char types stored as ASCII values -> direct conversion
int hexCharToInt(char c){
- if c from 0-9 -> return int(c) - int('0')
- if c from 'A' - 'F' -> return int(c) - int('A') + 10
	- ex: int('A') = 10 because 65 - 65 +10 = 10
}
